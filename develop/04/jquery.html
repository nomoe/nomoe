# 第四讲 检索模型-前端开发-jQuery版

# 1、回顾

在 jQuery 的版本中，我们已经开发了首页的搜索功能和新闻推荐功能，接下来就是完善搜索了，原本的搜索功能只包含模糊查询，因此我们还要再添加多两种搜索模式：全文索引和自定义分词查询，其中，自定义分词查询又分为 tfidf 和 bayes 查询，还需要提供升降排序选项，当然啦，我们也要开发分页功能，OK，接下来开搞~

最终代码：

// 初始化模态框
let modal = new Modal({
  width: 300,
  height: 200,
  title: "提示",
  text: "正在搜索中..."
});
modal.init();

// 后台接口路径
let searchUrl = {
  // 模糊查询
  fuzzy: 'http://localhost:8080/demo/article-fuzzy',
  // 全文索引
  fulltext: 'http://localhost:8080/demo/article-fulltext',
  // 自定义分词
  custom: 'http://localhost:8080/demo/article-custom'
}

// 搜索回调函数
function search() {
  // 获取搜索框的输入值，并且去除首尾空格
  let word = $('#search-input').val().trim();
  // 关键词为空或者空格
  if (word === "") {
    // 提示用户输入关键词再搜索
    alert('亲，您是想搜什么');
    return;
  }

  // 获取搜索选项
  let searchMode = $('input[name="searchMode"]:checked').val()
  let orderBy = $('input[name="orderBy"]:checked').val()
  let order = $('input[name="order"]:checked').val()

  // 显示模态框
  modal.show();
  // 发起网络请求调用后端搜索接口
  $.ajax({
    url: searchUrl[searchMode],
    method: 'get',
    data: {
      word: word,
      pageNumber: 1,
      pageSize: 10,
      orderBy: orderBy,
      order: order
    },
    success: function (result) {
      console.log(result)

      let html = '';
      let baseUrl = "news-view.html?id=";
      result.list.forEach(function (item) {
        html += searchResultTemplate.replace('{url}', baseUrl + item.id)
          .replace('{title}', item.title);
      })
      $('#search-result').html(html);

      // 渲染搜索结果信息
      let tips = resultInfoTemplate.replace('{word}', word)
        .replace('{duration}', result.duration)
        .replace('{size}', result.size);
      $('#tips').html(tips);

      // 隐藏模态框
      modal.hide()
    }
  })
}

// 搜索相关功能
let searchResultTemplate = $('#search-result-template').html();
let resultInfoTemplate = $('#real-tips').html();
$('#search-btn').click(search);
$('#search-input').keyup(function (e) {
  if (e.code === "NumpadEnter" || e.code === "Enter") {
    search();
  }
})

现在还差一个分页，为了方便，这次我就不手撕了，直接网上找插件即可，找完后再改造一波，现在，我把它封装成了 jk-paging，包含样式和逻辑文件，在首页直接引入即可~ 然后就可以开始使用了~

js:

// 初始化模态框
let modal = new Modal({
  width: 300,
  height: 200,
  title: "提示",
  text: "正在搜索中..."
});
modal.init();

// 后台接口路径
let searchUrl = {
  // 模糊查询
  fuzzy: 'http://localhost:8080/demo/article-fuzzy',
  // 全文索引
  fulltext: 'http://localhost:8080/demo/article-fulltext',
  // 自定义分词
  custom: 'http://localhost:8080/demo/article-custom'
}

// 搜索回调函数
function search() {
  // 获取搜索框的输入值，并且去除首尾空格
  let word = $('#search-input').val().trim();
  // 关键词为空或者空格
  if (word === "") {
    // 提示用户输入关键词再搜索
    alert('亲，您是想搜什么');
    return;
  }

  // 获取搜索选项
  let searchMode = $('input[name="searchMode"]:checked').val()
  let orderBy = $('input[name="orderBy"]:checked').val()
  let order = $('input[name="order"]:checked').val()

  // 对比关键词和搜索选项有没有发生改变，有的话当前页码和用户点击的页码都要重置
  if(word !== globalKeyword || searchMode !== globalSearchMode || orderBy !== globalOrderBy || order !== globalOrder) {
    currentPage = 1;
    pageNumber = 1;
  }

  // 显示模态框
  modal.show();
  // 发起网络请求调用后端搜索接口
  $.ajax({
    url: searchUrl[searchMode],
    method: 'get',
    data: {
      word: word,
      pageNumber: pageNumber,
      pageSize: 10,
      orderBy: orderBy,
      order: order
    },
    success: function (result) {
      // 渲染搜索结果
      let html = '';
      let baseUrl = "news-view.html?id=";
      result.list.forEach(function (item) {
        html += searchResultTemplate.replace('{url}', baseUrl + item.id)
          .replace('{title}', item.title);
      })
      $('#search-result').html(html);

      // 渲染搜索结果信息
      let tips = resultInfoTemplate.replace('{word}', word)
      .replace('{duration}', result.duration)
      .replace('{size}', result.size);
      $('#tips').html(tips);

      // 重新渲染分页
      new Paging("page", {
        nowPage: currentPage, // 当前页码
        pageNum: result.pages, // 总页码
        buttonNum: 5, //要展示的页码数量
        canJump: 0, // 是否能跳转。0：不显示（默认），1：显示
        showOne: 0, //只有一页时，是否显示。0：不显示,1：显示（默认）
        callback: function (num) {
          // 更新用户点击的页码
          pageNumber = num;
          // 更新当前页码
          currentPage = pageNumber;
          search();
        },
      });

      // 把搜索选项存储起来，以便下次搜索时对比
      globalSearchMode = searchMode;
      globalOrderBy = orderBy;
      globalOrder = order;
      globalKeyword = word;
      // 隐藏模态框
      modal.hide()
    }
  })
}

// 搜索相关功能
let searchResultTemplate = $('#search-result-template').html();
let resultInfoTemplate = $('#real-tips').html();
let currentPage = 1;
let pageNumber = 1;
let globalSearchMode, globalOrderBy, globalOrder, globalKeyword;
$('#search-btn').click(search);
$('#search-input').keyup(function (e) {
  if (e.code === "NumpadEnter" || e.code === "Enter") {
    search();
  }
})
