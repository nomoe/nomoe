# 重要：请先浏览jQuery版本的教程

# 2、新建 Vue 实例对象

在 js 目录下新建 index.js，同时在 index.html 中引入，在 index.js 中新建一个 vue实例

// 关闭Vue生产环境提示
Vue.config.productionTip = false;

new Vue({
  el: '#app',
  data: {
    // 搜索结果信息
    tips: {
      word: '关键词',
      duration: 1024,
      size: 1024
    },
    // 搜索结果列表
    searchResultList: [{
      id: 1,
      title: '搜索结果1'
    },{
      id: 2,
      title: '搜索结果2'
    }],
    // 新闻推荐列表
    recommendNewsList: [{
      id: 1,
      title: '推荐新闻1'
    },{
      id: 2,
      title: '推荐新闻2'
    }]
  }
})

# index.html

<!DOCTYPE html>
<html lang="zh-CN">
<head>
 <meta charset="UTF-8">
 <title>新闻推荐系统</title>
 <link rel="stylesheet" href="css/normalize.css">
 <link rel="stylesheet" href="css/index.css">
 <script src="js/vue.js"></script>
 <style>
  [v-cloak] {
   display: none;
  }
 </style>
</head>
<body>
<!-- 页面容器 -->
<div id="app">
 <!-- 头部 -->
 <div class="header">
  <h3 class="header-title">NewsDemo</h3>
 </div>

 <!-- 主体 -->
 <div class="main clearfix">

  <!-- 左栏 -->
  <div class="main-left">
   <!-- 搜索框 -->
   <div class="search">
    <label>
     <input type="text" class="search-input">
    </label>
    <button class="search-btn">
     搜索
    </button>
   </div>

   <p class="tips" v-cloak>搜索关键词：<span>{{tips.word}}</span>，耗时：<span>{{tips.duration}}</span>毫秒，返回：<span>{{tips.size}}</span>条新闻</p>

   <div class="container">
    <h3 class="news-list-title">搜索结果</h3>
    <ul>
     <li class="news-list-item" v-for="news in searchResultList" :key="news.id">
      <a :href="`news-view.html?id=${news.id}`" target="_blank">
       {{news.title}}
      </a>
     </li>
    </ul>
   </div>
  </div>

  <!-- 右栏 -->
  <div class="main-right">
   <div class="container">
    <h3 class="news-list-title">新闻推荐</h3>
    <ul>
     <li class="news-list-item" v-for="news in recommendNewsList" :key="news.id">
      <a :href="`news-view.html?id=${news.id}`" target="_blank">
       {{news.title}}
      </a>
     </li>
    </ul>
   </div>
  </div>
 </div>

 <!-- 页脚 -->
 <div class="footer">
  <p>Copyright 2021-2077 Jack All rights reserved.</p>
 </div>
</div>

<script src="js/index.js"></script>
</body>
</html>
