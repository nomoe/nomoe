打开浏览器，发现报错了，很明显，稍微有点经验的童鞋应该看出这是出现跨域问题了，跨域是因为浏览器的安全机制引起的，如果当前url 的协议，域名或端口任意一个不一样都会触发跨域，具体详情可百度，在这里只谈解决方法，当然，解决方法有很多，我们选择用后端 CORS （跨域资源共享，W3C 标准）的方式，这也是推荐使用的方式

所以我们在后端项目的 filter 包下新建一个类 CorsFilter ，采用注解的方式配置，默认拦截所有路径

package com.example.demo.filter;

import javax.servlet.*;
import javax.servlet.annotation.WebFilter;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

@WebFilter("/*")
public class CorsFilter implements Filter {
  @Override
  public void init(FilterConfig filterConfig) {

  }

  @Override
  public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain) throws IOException, ServletException {
    HttpServletRequest request = (HttpServletRequest) servletRequest;
    HttpServletResponse response = (HttpServletResponse) servletResponse;
    String originalURL = request.getHeader("Origin");
    if (originalURL != null) {
      response.addHeader("Access-Control-Allow-Origin", originalURL);
    }
    response.addHeader("Access-Control-Allow-Credentials", "true");
    filterChain.doFilter(request, response);
  }

  @Override
  public void destroy() {

  }
}

success: function (result) {
  // 获取模板字符串
  let template = $('#search-result-template').html();
  // 存放列表总的html代码
  let html ='';
  // 新闻url前缀
  let baseUrl = "news-view.html?id=";
  // 遍历结果集
  result.list.forEach(function(item) {
    // 替换模板字符串为具体内容，拼接模板
    html += template.replace('{url}', baseUrl + item.id)
      .replace('{title}', item.title);
  })
  // 将拼接后的html代码放置到相应位置
  $('#search-result').html(html)
}

function getQueryVariable(name) {
  let query = window.location.search.substring(1);
  let params = query.split("&");
  for (let i = 0; i < params.length; i++) {
    let pair = params[i].split("=");
    if (pair[0] === name) {
      return pair[1];
    }
  }
  return false;
}
